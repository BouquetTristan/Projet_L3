def initialize (diff, nbF)
        @difficulte = diff
        case diff
        when "Facile" then
                @scoreInitial = 180
        when "Moyenne" then
                @scoreInitial = 360
        when "Difficile" then
                @scoreInitial = 720
        end
        @nbFeuilles = nbF
end
