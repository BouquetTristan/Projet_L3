def detruitMoi
  @window.each{ |children|
    unless children.class.eql?(Header) then
      children.destroy
    end
  }
end

def supprimeMoi
  @window.each{ |children|
    unless children.class.eql?(Header) then
      @window.remove(children)
    end
  }
end
